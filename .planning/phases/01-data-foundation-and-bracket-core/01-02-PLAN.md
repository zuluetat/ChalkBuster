**js/state.js** — Single source of truth. Implement exactly these exports:

export const state = {
  teams: {},
  slots: {},
  firstFour: {},
  picks: {},
  activeRegion: "East",
};

export function setPick(gameId, teamId) {
  state.picks[gameId] = teamId;
  clearDownstream(gameId);
  savePicksToStorage();
}

export function clearDownstream(gameId) {
  // Find all slots whose top or bot source references gameId as a winner
  const downstream = Object.values(state.slots).filter(slot =>
    slot.top?.game === gameId || slot.bot?.game === gameId
  );
  downstream.forEach(slot => {
    if (state.picks[slot.id]) {
      delete state.picks[slot.id];
      clearDownstream(slot.id);
    }
  });
}

export function savePicksToStorage() {
  try {
    localStorage.setItem("chalkbuster-picks-v1", JSON.stringify(state.picks));
  } catch (e) {
    console.warn("[ChalkBuster] Could not save picks to localStorage:", e);
  }
}

export function loadPicksFromStorage() {
  try {
    const saved = localStorage.getItem("chalkbuster-picks-v1");
    if (saved) {
      state.picks = JSON.parse(saved);
    }
  } catch (e) {
    console.warn("[ChalkBuster] Could not load picks from localStorage:", e);
    state.picks = {};
  }
}

**js/bracket.js** — Bracket renderer and click handler. Imports from state.js only. Never imports from app.js.

Must implement:

function resolveTeam(source) — resolves which team object occupies a slot source:
- if source.type === "seed": find team where t.seed === source.seed && t.region === source.region
- if source.type === "first_four": get ff = state.firstFour[source.ff_id]; if ff.resolved && ff.winner return state.teams[ff.winner]; else return null
- if source.type === "winner": get winnerId = state.picks[source.game]; if winnerId return state.teams[winnerId]; else return null
- returns null for unresolvable

export function renderBracket() — renders all 4 regions into #bracket-container:
- Clears #bracket-container innerHTML
- For each region ["East","South","Midwest","West"]:
  - Creates div.bracket-region with data-region attribute
  - Adds .active class if region === state.activeRegion
  - Groups region slots by round: R64 -> R32 -> S16 -> E8 (4 columns)
  - Renders each round as div.round-column
  - Renders each slot as a matchup card (see card structure below)
- Also renders Final Four and Championship region (region "FinalFour" + "Championship" slots)

Matchup card HTML structure for a slot:
  <div class="matchup-card" data-game="SLOT_ID">
    <div class="team-slot [picked?] [tbd?]" data-pos="top" data-team="TEAM_ID_or_empty">
      <span class="seed-badge">SEED</span>
      <span class="team-name">TEAM_NAME or TBD</span>
    </div>
    <div class="team-slot [picked?] [tbd?]" data-pos="bot" data-team="TEAM_ID_or_empty">
      <span class="seed-badge">SEED</span>
      <span class="team-name">TEAM_NAME or TBD</span>
    </div>
  </div>

Rules:
- data-team is set to team.id if team is resolved, empty string if TBD
- .tbd class on team-slot if resolveTeam returns null for that position
- .picked class if state.picks[slot.id] === team.id
- Do NOT attach click handlers to TBD slots (when resolveTeam returns null)

export function updateSlotAndDownstream(gameId) — targeted DOM update:
- Calls updateSlotDOM(gameId)
- Finds downstream slots (slots where top.game===gameId or bot.game===gameId)
- Calls updateSlotAndDownstream recursively on each

function updateSlotDOM(gameId) — updates a single slot card in the DOM:
- Selects element: document.querySelector("[data-game=\"" + gameId + "\"]")
- If not found, return (slot not in current visible region, safe to skip)
- Calls resolveTeam for top and bot sources
- Updates top and bot team-slot elements: text, data-team, .tbd class, .picked class
- Re-attaches click handler (or removes and re-adds)

export function initBracketHandlers() — sets up click delegation:
- Add single click listener on #bracket-container
- On click, find closest .team-slot ancestor
- Read data-game from closest .matchup-card ancestor, data-team from .team-slot
- If data-team is empty string, return (TBD — do not pick)
- If data-team is already the current pick for that game, return (no re-pick needed)
- Call setPick(gameId, teamId) from state.js
- Call updateSlotAndDownstream(gameId) to refresh DOM

import { state, loadPicksFromStorage } from "./state.js";
import { renderBracket, initBracketHandlers } from "./bracket.js";
import { renderRegionTabs } from "./regions.js";

async function init() {
  try {
    const [teamsData, bracketData, ffData] = await Promise.all([
      fetch("./data/teams.json").then(r => r.json()),
      fetch("./data/bracket.json").then(r => r.json()),
      fetch("./data/first_four.json").then(r => r.json()),
    ]);

    state.teams = teamsData.teams;
    state.slots = bracketData.slots;
    state.firstFour = ffData.first_four;

    loadPicksFromStorage();

    renderRegionTabs();
    renderBracket();
    initBracketHandlers();

  } catch (err) {
    console.error("[ChalkBuster] init failed:", err);
    document.getElementById("app").textContent =
      "Failed to load bracket data. Is the dev server running? Try: python3 -m http.server 8000";
  }
}
